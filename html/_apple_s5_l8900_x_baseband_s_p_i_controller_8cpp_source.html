113 IOReturn AppleS5L8900XBasebandSPIController::loadConfiguration( void )
114 {
115  SPI_TRACE( DEBUG_SPEW );
116 
117  IOReturn ret;
118  ret = super::loadConfiguration();
119  if ( ret != kIOReturnSuccess )
120  {
121  return ret;
122  }
123 
124  _spiConReg = kS5L8900XSPCONClkSelNClk | kS5L8900XSPCONClkEnable | kS5L8900XSPCONMasterMode;
125 
126  if (fConfiguration.spiClockPolarity) _spiConReg |= kS5L8900XSPCONClkActiveLow;
127  if (fConfiguration.spiClockPhase) _spiConReg |= kS5L8900XSPCONClkMissingLast;
128  if (!fConfiguration.spiBitFirst) _spiConReg |= kS5L8900XSPCONBitOrderLSB;
129 
130  UInt32 sclkPeriod = fConfiguration.spiClockPeriod;
131  {
132  UInt32 period = getOverrideClockPeriod();
133  if ( period != 0 )
134  {
135  sclkPeriod = period;
136  SPI_LOG( DEBUG_CRITICAL, "Overriding SCLK to %uns\n", (unsigned)period );
137  }
138  else
139  {
140  SPI_LOG( DEBUG_SPEW, "SCLK period %uns\n", (unsigned)sclkPeriod );
141  }
142  }
143 
144  SPI_LOG( DEBUG_SPEW, "polarity %p, phase %p\n",
145  (void *)fConfiguration.spiClockPolarity, (void *)fConfiguration.spiClockPhase );
146 
147  switch ( fConfiguration.spiWordSize )
148  {
149  case 8:
150  _spiWordSize = 0;
151  break;
152  case 16:
153  _spiWordSize = 1;
154  break;
155  case 32:
156  _spiWordSize = 2;
157  break;
158  default:
159  SPI_ASSERT( fConfiguration.spiWordSize, ==, 8 );
160  }
161 
162  _spiConReg |= _spiWordSize << kS5L8900XSPCONWordSizeShift;
163  _spiWordSize = 1 << _spiWordSize;
164 
165 #if SPI_WORD32
166  // Request in the unit of every one buffer. The documentation
167  // states one byte however it is wrong.
168  // TODO Is it worse to poll the last 4 buffers
169  _spiConReg |= kS5L8900XSPCONDMAReqOne | kS5L8900XSPCONDataSwap;
170 #endif
171 
172  // SPI Clock Value = Cell Clock / prescale
173  _spiPreReg = (((UInt64)fNCLKFrequency * sclkPeriod) + 1000000000ULL - 1) / 1000000000ULL;
174  SPI_LOG( DEBUG_SPEW, "NCLK %p\t Prescaler %p\n", (void *)fNCLKFrequency, (void *)_spiPreReg );
175 
176  // SPI Inter-byte Delay = PCLK period * delay cycles
177  _spiIDDReg = (((UInt64)fPCLKFrequency * fConfiguration.spiWordDelay) + 1000000000ULL - 1) / 1000000000ULL;
178 
179  SPI_LOG( DEBUG_SPEW, "SPCON: %p\tWord Size: %p\n", (void *)_spiConReg, (void *)_spiWordSize );
180  SPI_LOG( DEBUG_SPEW, "SPPRE: %p\tSPIDD: %p\n", (void *)_spiPreReg, (void *)_spiIDDReg );
181 
182  return kIOReturnSuccess;
183 }

185 IOReturn AppleS5L8900XBasebandSPIController::configureHardware( UInt32 dmaFlags, IOByteCount size )
186 {
187  SPI_TRACE( DEBUG_SPEW );
188 
189  SPI_LOG( DEBUG_SPEW, "Flags = %p, size = %u\n", (void *)dmaFlags, (unsigned)size );
190 
191  enableClock(true);
192  // Reset & Clear Interrupts
193  _spiClkConReg = kS5L8900XSPCLKCONOn;
194 
195  // disable the clock and flush
196  writeReg32(kS5L8900XSPCLKCON, kS5L8900XSPCLKCONFlushTx | kS5L8900XSPCLKCONFlushRx);
197 
198  // clear the pin control register
199  writeReg32(kS5L8900XSPPIN, 0);
200  // set the inter-word data delay
201  writeReg32(kS5L8900XSPIDD, _spiIDDReg);
202 
203  // clear the count
204  writeReg32(kS5L8900XSPCNT, 0);
205 
206  // set the baud rate prescaler
207  writeReg32(kS5L8900XSPPRE, _spiPreReg);
208 
209  // clear all interrupts
210  writeReg32(kS5L8900XSPSTA, kS5L8900XSPSTAIntAll);
211 
212  // set SPCON
213  writeReg32(kS5L8900XSPCON, _spiConReg );
214 
215  size = size >> (_spiWordSize >> 1);
216 
217  // write the count register
218  writeReg32(kS5L8900XSPCNT, size);
219 
220  UInt32 spiConReg = _spiConReg | kS5L8900XSPCONSPIModeDMA;
221  if ( dmaFlags & kTxDMA )
222  {
223  spiConReg |= kS5L8900XSPCONFullDuplex;
224  }
225  else
226  {
227  spiConReg |= kS5L8900XSPCONAGDMode;
228  }
229  // enable DMA and set garbage tx frame mode (if needed)
230  writeReg32(kS5L8900XSPCON, spiConReg );
231 
232  return kIOReturnSuccess;
233 }

268 {
269  bool wasClockEnabled = clockEnabled();
270  // Enable the SPI clock temporarily for snapshot, if not already enabled
271  if (!wasClockEnabled)
272  {
273  enableClock(true);
274  }
275 
276  *snapshot = new AppleS5L8900XBasebandSPIControllerSnapshot( comment,
277  resetDetectState,
278  readReg32(kS5L8900XSPCLKCON),
279  readReg32(kS5L8900XSPCON),
280  readReg32(kS5L8900XSPSTA),
281  readReg32(kS5L8900XSPCNT) );
282 
283  // Disable SPI clock temporarily for snapshot, if enabled just for duration of snapshot
