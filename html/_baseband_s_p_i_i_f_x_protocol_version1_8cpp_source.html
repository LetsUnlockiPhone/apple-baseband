756 
757 #if SPI_PEDANTIC_ASSERT
758  SPI_ASSERT( command, !=, NULL );
759 #endif
760 
761  command->dumpFrame( DEBUG_SPEW );
762 
763  // If we were low on Tx commands, mark that we want to notify our
764  // client that a Tx buffer is now free
765  if ( getNumTxUnusedCommands() == 1 )
766  {
767  callbackFlags |= kBasebandSPIReturnTxReady;
768  }
769 
770  // TODO: this might be redundant
771  if ( fCommandQueue->getSize() )
772  {
773  prepareFlags |= kPrepareInitiate;
774  }
775 
776  addTxUnusedCommand( command );
777  }
778 
779  {
780  // we always go in here since we always prepare for an Rx
781 
782  command = OSDynamicCast( BasebandSPICommandVersion1, receivedCommand );
783 
784  SPI_ASSERT( command, !=, NULL );
785 
786  command->dumpFrame( DEBUG_SPEW );
787 
788  IOByteCount payloadLength = command->getPayloadLength();
789 
790  if ( payloadLength > getFramePayloadSize() )
791  {
792  if ( payloadLength < BasebandSPICommandVersion1::kSPIHeaderMaxDataSize )
793  {
794  // the payload length is greater than what we expected, but the field isn't all 1's (in binary)
795  // so dump the frame as we shouldn't get here
796  SPI_LOG( DEBUG_CRITICAL, "Recieved frame of payload size %u, frame size %u\n",
797  payloadLength, getFramePayloadSize() );
798 
799  command->dumpFrame( DEBUG_CRITICAL );
800  }
801  else
802  {
803  // the frame was all 1's (in binary) and thus the slave had nothing to send us
804  }
805 
806  addRxUnusedCommand( command );
807 
808 
809  // ramp up the controller agin
810  prepareTransferIfNeeded( prepareFlags );
811  }
812  else
813  {
814 
815  // set our state to only the master flags as we be reading in the new slave flags
816  fStateFlags = fStateFlags & kMasterStateFlagMask;
817 
818  // go through what we received, set flags in our state accordingly
819  if ( command->indicatesMoreToTransfer() )
820  {
821 #if SPI_SRDY_DEBUG
822  BasebandSPILogger::logGeneral( "SMORE", 1, isSRDYStateHigh() );
823 #else
824  BasebandSPILogger::logGeneral( "SMORE", 1 );
825 #endif
826  fStateFlags |= kFlagSlaveMoreAsserted;
827  prepareFlags |= kPrepareInitiate;
828  }
829 
830  if ( command->indicatesFlowControl() )
831  {
832 #if SPI_SRDY_DEBUG
833  BasebandSPILogger::logGeneral( "SFLOW", 1, isSRDYStateHigh() );
834 #else
835  BasebandSPILogger::logGeneral( "SFLOW", 1 );
836 #endif
837  fStateFlags |= kFlagSlaveFlowControlAsserted;
838  }
839 
840  SPI_LOG( DEBUG_SPEW, "State flags %p\n", fStateFlags );
841 
842  // ramp up the controller again
843  // we do this here because if there was another pending SRDY, it can transfer
844  // while the client processes the frame we received
845  prepareTransferIfNeeded( prepareFlags );
846 
847  if ( payloadLength )
848  {
849  // if there was an actual payload, mark that this is a real Rx
850  callbackFlags |= kBasebandSPIReturnRxReady;
851 
852  BasebandSPILogger::logFrameReceived( payloadLength + sizeof( BasebandSPICommandVersion1::SPIHeader ), command );
853 
854  // clear the memory descriptor of the command and put it back in the free pool
855  command->clearMemoryDescriptor();
856  addRxFreeCommand( command );
857 
858  // do the actual callback
859  BasebandSPILogger::logReceiveCallbackBegin();
860  fDevice->invokeCompletionCallback( command->getCookie(), callbackFlags,
861  sizeof( BasebandSPICommandVersion1::SPIHeader ), payloadLength );
862  BasebandSPILogger::logReceiveCallbackEnd();
863  }
864  else
865  {
866  // this was a header-only frame, recycle the command and use it for another receive
867  BasebandSPILogger::logFrameReceived( sizeof( BasebandSPICommandVersion1::SPIHeader ), command );
868 
869  addRxUnusedCommand( command );
870  }
871  }
872  }
873 
874  return kIOReturnSuccess;
875 }
