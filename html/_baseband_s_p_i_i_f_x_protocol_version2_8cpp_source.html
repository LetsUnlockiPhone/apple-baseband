785 IOReturn BasebandSPIIFXProtocolVersion2::handleTransferComplete( IODMACommand * receivedCommand, IODMACommand * sendCommand )
786 {
787  SPI_TRACE( DEBUG_SPEW );
788 
789  BasebandSPICommandVersion2 * command;
790 
791  SPI_LOG( DEBUG_SPEW, "Transfer flags = %p\n", fTransferFlags );
792 
793  // if we didn't indicate MORE last time, lower MRDY
794  if ( !( fLastStateFlags & kFlagMasterMoreAsserted ) )
795  {
796  lowerMRDY();
797  }
798 
799  stopSRDYTimeoutTimer();
800 
801  fController->beginBlockInterrupt();
802  fController->unconfigure();
803 
804  bool missed = fController->endBlockInterrupt();
805 
806  UInt32 prepareFlags;
807 
808  // if we missed an SRDY while we blocked the interupt
809  // or we missed it because our transfer was active, mark that we
810  // want to initiate and start a transfer
811  if ( missed || ( fTransferFlags & kFlagMissedSRDY ) )
812  {
813  prepareFlags = kPrepareInitiate | kPrepareStart;
814  }
815  else
816  {
817  prepareFlags = 0;
818  }
819 
820  fTransferFlags = 0;
821 
822  UInt32 callbackFlags = 0;
823 
824  if ( sendCommand )
825  {
826  // only go in here if we sent a valid frame (either with real payload or header-only)
827 
828  command = OSDynamicCast( BasebandSPICommandVersion2, sendCommand );
829 
830 #if SPI_PEDANTIC_ASSERT
831  SPI_ASSERT( command, !=, NULL );
832 #endif
833 
834  command->dumpFrame( DEBUG_SPEW );
835 
836  // If we were low on Tx commands, mark that we want to notify our
837  // client that a Tx buffer is now free
838  if ( getNumTxUnusedCommands() == 1 )
839  {
840  callbackFlags |= kBasebandSPIReturnTxReady;
841  }
842 
843 
844  // <rdar://problem/8730644> AP might initiate Tx transfers unnecessarily
845  /*
846  if ( fCommandQueue->getSize() )
847  {
848  prepareFlags |= kPrepareInitiate;
849  }
850  */
851 
852  addTxUnusedCommand( command );
853  }
854 
855  {
856  // we always go in here since we always prepare for an Rx
857 
858  command = OSDynamicCast( BasebandSPICommandVersion2, receivedCommand );
859 
860  SPI_ASSERT( command, !=, NULL );
861 
862  command->dumpFrame( DEBUG_SPEW );
863 
864  IOByteCount payloadLength = command->getPayloadLength();
865 
866  if ( payloadLength > getFramePayloadSize() )
867  {
868  if ( payloadLength < BasebandSPICommandVersion2::kSPIHeaderMaxDataSize )
869  {
870  // the payload length is greater than what we expected, but the field isn't all 1's (in binary)
871  // so dump the frame as we shouldn't get here
872  SPI_LOG( DEBUG_CRITICAL, "Recieved frame of payload size %u, frame size %u\n",
873  payloadLength, getFramePayloadSize() );
874 
875  command->dumpFrame( DEBUG_CRITICAL );
876  }
877  else
878  {
879  // the frame was all 1's (in binary) and thus the slave had nothing to send us
880  }
881 
882  addRxUnusedCommand( command );
883 
884 
885  // ramp up the controller agin
886  prepareTransferIfNeeded( prepareFlags );
887  }
888  else
889  {
890  // add the number of credits we have
891  fCurrentTxCredits += command->getGrantedCredits();
892 #if SPI_SRDY_DEBUG
893  BasebandSPILogger::logGeneral( "CRDGTx", (command->getGrantedCredits() << 16 ) | fCurrentTxCredits, isSRDYStateHigh() );
894 #else
895  BasebandSPILogger::logGeneral( "CRDGTx", (command->getGrantedCredits() << 16 ) | fCurrentTxCredits );
896 #endif
897 
898  // set our state to only the master flags as we be reading in the new slave flags
899  fStateFlags = fStateFlags & kMasterStateFlagMask;
900 
901  // go through what we received, set flags in our state accordingly
902  if ( command->indicatesMoreToTransfer() )
903  {
904 #if SPI_SRDY_DEBUG
905  BasebandSPILogger::logGeneral( "SMORE", 1, isSRDYStateHigh() );
906 #else
907  BasebandSPILogger::logGeneral( "SMORE", 1 );
908 #endif
909  fStateFlags |= kFlagSlaveMoreAsserted;
910  prepareFlags |= kPrepareInitiate | kPrepareStart;
911  }
912 
913  // if there is an error reported, throw up an error
914  if ( command->indicatesRxError() )
915  {
916  fController->reportError( kBasebandSPIReturnProtocolError, "Rx Error" );
917  }
918 
919  SPI_LOG( DEBUG_SPEW, "State flags %p\n", fStateFlags );
920 
921  // ramp up the controller again
922  // we do this here because if there was another pending SRDY, it can transfer
923  // while the client processes the frame we received
924  prepareTransferIfNeeded( prepareFlags );
925 
926  if ( payloadLength )
927  {
928 #if SPI_PEDANTIC_ASSERT
929  SPI_ASSERT( fSlaveRxCredits, >, 0 );
930 #endif
931 
932  // we received real data, so the baseband consumed a credit
933  fSlaveRxCredits--;
934  fCurrentRxCredits--;
935 #if SPI_SRDY_DEBUG
936  BasebandSPILogger::logGeneral( "CRDDRx", ( fSlaveRxCredits << 16 ) | fCurrentRxCredits, isSRDYStateHigh() );
937 #else
938  BasebandSPILogger::logGeneral( "CRDDRx", ( fSlaveRxCredits << 16 ) | fCurrentRxCredits );
939 #endif
940  SPI_LOG( DEBUG_SPEW, "Slave consumed %u credits\n", fCurrentRxCredits );
941 
942  // if there was an actual payload, mark that this is a real Rx
943  callbackFlags |= kBasebandSPIReturnRxReady;
944 
945  BasebandSPILogger::logFrameReceived( payloadLength + sizeof( BasebandSPICommandVersion2::SPIHeader ), command );
946 
947  // clear the memory descriptor of the command and put it back in the free pool
948  command->clearMemoryDescriptor();
949  addRxFreeCommand( command );
950 
951  // do the actual callback
952  BasebandSPILogger::logReceiveCallbackBegin();
953  fDevice->invokeCompletionCallback( command->getCookie(), callbackFlags,
954  sizeof( BasebandSPICommandVersion2::SPIHeader ), payloadLength );
955  BasebandSPILogger::logReceiveCallbackEnd();
956  }
957  else
958  {
959  // this was a header-only frame, recycle the command and use it for another receive
960  BasebandSPILogger::logFrameReceived( sizeof( BasebandSPICommandVersion2::SPIHeader ), command );
961 
962  addRxUnusedCommand( command );
963  }
964  }
965  }
966 
967  return kIOReturnSuccess;
968 }
