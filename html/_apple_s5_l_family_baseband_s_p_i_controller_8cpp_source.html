102 IOReturn AppleS5LFamilyBasebandSPIController::loadConfiguration( void )
103 {
104  IOReturn ret = super::loadConfiguration();
105 
106  if ( ret != kIOReturnSuccess )
107  {
108  return ret;
109  }
110 
111  if ( !fConfiguration.spiClockPeriod )
112  {
113  SPI_LOG( DEBUG_CRITICAL, "SPI clock period == 0\n" );
114  return kIOReturnBadArgument;
115  }
116 
117  if ((((fNCLKFrequency * fConfiguration.spiClockPeriod) + 1000000000ULL - 1) / 1000000000ULL) > 2048) {
118  SPI_LOG( DEBUG_CRITICAL, "Invalid SPI clock period %u\n", fConfiguration.spiClockPeriod);
119  return kIOReturnBadArgument;
120  }
121 
122  // S5L8920X supports only 8, 16 & 32 bit words
123  switch ( fConfiguration.spiWordSize )
124  {
125  case 8:
126  case 16:
127  case 32:
128  break;
129  default:
130  SPI_LOG( DEBUG_CRITICAL, "Invalid SPI word size %u\n", fConfiguration.spiWordSize);
131  return kIOReturnBadArgument;
132  }
133 
134  SPI_TRACE( DEBUG_SPEW );
135 
136  // get our clock frequencies
137  fPCLKFrequency = fProvider->getClockFrequency("pclk");
138  if (!fPCLKFrequency) return false;
139 
140  fNCLKFrequency = fProvider->getClockFrequency("nclk");
141  if (!fNCLKFrequency) return false;
142 
143  // get our memory map and base address
144  fMemoryMap = fProvider->mapDeviceMemoryWithIndex(0);
145  if (!fMemoryMap) return false;
146  fBaseAddress = fMemoryMap->getVirtualAddress();
147 
148  SPI_LOG( DEBUG_INFO, "baseAddress %p, physical %p\n", fBaseAddress,
149  fMemoryMap->getPhysicalAddress() );
150 
151  return kIOReturnSuccess;
152 }
