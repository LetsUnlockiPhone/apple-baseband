453 {
454  SPI_TRACE( DEBUG_INFO );
455 
456  SPI_ASSERT( fInLowPower, ==, false );
457  fEnteringLowPower = true;
458 
459  switch( fState )

461  case kStateActive:

476  if ( fState == kStateActive )

478  fStateBeforeLowPower = fState;

481  }
482  }
483  // fall through
484 
485  case kStateInactive:
486  fStateBeforeLowPower = fState;
487  break;
488 
489  }
490 
491  fEnteringLowPower = false;
492  fInLowPower = true;
493 
494  return kIOReturnSuccess;

1053 bool BasebandSPIController::reportError( BasebandSPIReturn err, const char * details )
1054 {
1055  switch( fState )
1056  {
1057  case kStateInactive:
1058  case kStateDeactivating:
1059  // if we're not enabled, then don't throw up an error.
1060  return false;
1061 
1062  default:
1063  break;
1064  }
1065 
1066  {
1067  char buff[256];
1068 
1069  SPIsnprintf( buff, sizeof( buff ), "%s/%p", details, err );
1070 
1071  IOReturn ret = getNewSnapshot( buff );
1072 
1073  if ( ret != kIOReturnSuccess )
1074  {
1075  SPI_LOG( DEBUG_CRITICAL, "Failed to create new snapshot, ret = %p\n", ret );
1076  }
1077  else
1078  {
1079  SPI_ASSERT( fSavedSnapshot, !=, NULL );
1080 
1081  fSavedSnapshot->copyAsString( buff, sizeof( buff ) );
1082 
1083  SPI_LOG( DEBUG_CRITICAL, "Error: %s\n", buff );
1084  }
1085  }
1086 
1087  bool ret;
1088  if ( fSavedSnapshot )
1089  {
1090  if ( fDevice )
1091  {
1092  fDevice->reportError( err );
1093  }
1094 
1095  ret = true;
1096  }
1097  else
1098  {
1099  // Delete the snapshot if the error is not being reported to upper layers
1100  if ( fSavedSnapshot )
1101  {
1102  delete fSavedSnapshot;
1103  fSavedSnapshot = NULL;
1104  }
1105  ret = false;
1106  }
1107 
1108 #if SPI_ASSERT_GPIO_ON_ERROR
1109  AppleARMFunction * function = AppleARMFunction::withProvider(fProvider, "function-fail_gpio");
1110 
1111  if ( function )
1112  {
1113  UInt32 value;
1114  IOReturn ret;
1115 
1116  SPI_LOG( DEBUG_CRITICAL, "Triggering GPIO\n" );
1117 
1118  value = 1;
1119  ret = function->callFunction( &value );
1120  SPI_ASSERT( kIOReturnSuccess, ==, ret );
1121 
1122  IOSleep( 20 );
1123 
1124  value = 0;
1125  ret = function->callFunction( &value );
1126  SPI_ASSERT( kIOReturnSuccess, ==, ret );
1127 
1128  IOSleep( 100 );
1129 
1130  value = 1;
1131  ret = function->callFunction( &value );
1132  SPI_ASSERT( kIOReturnSuccess, ==, ret );
1133 
1134 
1135 bail:
1136  function->release();
1137  }
1138 #endif
1139 
1140  return ret;
1141 }
