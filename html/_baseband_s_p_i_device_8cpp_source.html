93 bool BasebandSPIDevice::start( IOService * provider )
94 {
95  SPI_TRACE( DEBUG_SPEW );
96 
97  //PE_enter_debugger( "blah" );
98 
99  bool ret = super::start( provider );
100  SPI_ASSERT( ret, ==, true );
101 
102  fController = OSDynamicCast(BasebandSPIController, provider);
103  if (!fController){
104  SPI_LOG( DEBUG_CRITICAL, "Controller seems invalid\n" );
105  return false;
106  }
107 
108  fCommandGate = IOCommandGate::commandGate( this );
109  if ( fCommandGate == NULL )
110  {
111  SPI_LOG( DEBUG_CRITICAL, "Failed to create command gate\n" );
112  return false;
113  }
114 
115  SPI_ASSERT( getWorkLoop(), !=, NULL );
116 
117  IOReturn result;
118 
119  result = getWorkLoop()->addEventSource( fCommandGate );
120 
121  if ( result != kIOReturnSuccess )
122  {
123  fCommandGate->release();
124  fCommandGate = NULL;
125  SPI_LOG( DEBUG_CRITICAL, "Failed to add command gate to workloop, ret = %p\n", result );
126  return false;
127  }
128 
129  fProtocol = BasebandSPIProtocol::create( fController, this );
130  if ( !fProtocol )
131  {
132  SPI_LOG( DEBUG_CRITICAL, "Creating protocol failed\n" );
133  return false;
134  }
135 
136  result = initPM();
137  if ( result != kIOReturnSuccess )
138  {
139  SPI_LOG( DEBUG_CRITICAL, "Couldn't initialize PM\n" );
140  }
141 
142  registerService();
143 
144  return true;
145 }

210 IOReturn BasebandSPIDevice::activate( void * target, BasebandSPICompletionAction action )
211 {
212  SPI_TRACE( DEBUG_INFO );
213 
214  // while this flag is set, we can't activate. We could get in here
215  // if we had previuosly failed to complete a proper low power sequence
216  // and then CommCenter or someone re-opened us while the system was still
217  // in the process of entering sleep
218  while( fLowPowerFlags & kFlagSystemInLowPower )
219  {
220  IOReturn ret;
221  ret = fCommandGate->commandSleep( &fLowPowerFlags, THREAD_UNINT );
222 
223  if ( ret != kIOReturnSuccess )
224  {
225  SPI_LOG( DEBUG_CRITICAL, "Couldn't wait for system wake\n" );
226  return ret;
227  }
228  }
229 
230  switch ( fState )
231  {
232  case kStateInactive:
233  {
234  IOReturn ret;
235 
236  ret = fController->activate( this, fProtocol );
237  SPI_ASSERT_SUCCESS( ret );
238 
239  fState = kStateActive;
240 
241  fCompletionTarget = target;
242  fCompletionAction = action;
243 
244  fProtocol->activate();
245  SPI_ASSERT_SUCCESS( ret );
246 
247  return kIOReturnSuccess;
248  }
249 
250  default:
251  SPI_LOG(DEBUG_CRITICAL, kInvalidState, stateAsString( fState ) );
252  return kIOReturnNotPermitted;
253  }
254 }

316 IOReturn BasebandSPIDevice::enterLowPower( void )
317 {
318  SPI_TRACE( DEBUG_INFO );
319 
320  fLowPowerFlags &= ~kFlagLowPowerPreEnter;
321  switch ( fState )
322  {
323  case kStateActive:
324  {
325  if ( !( fLowPowerFlags & kFlagLowPowerAllowed ) )
326  {
327  fController->reportError( kBasebandSPIReturnPMError, "child driver did not allow low power" );
328  }
329  else
330  {
331  fLowPowerFlags &= ~ kFlagLowPowerAllowed;
332  }
333  }
334  // fall through
335  case kStateInactive:
336  {
337  IOReturn ret = fController->enterLowPower();
338  SPI_ASSERT_SUCCESS( ret );
339 
340  ret = fProtocol->enterLowPower();
341  SPI_ASSERT_SUCCESS( ret );
342 
343  fStateBeforeLowPower = fState;
344  fState = kStateLowPower;
345 
346 
347  fLowPowerFlags |= kFlagSystemInLowPower;
348  }
349  SPI_LOG( DEBUG_INFO, "In Low Power\n" );
350  return kIOReturnSuccess;
351 
352  default:
353  SPI_LOG(DEBUG_CRITICAL, kInvalidState, stateAsString( fState ) );
354  return kIOReturnNotPermitted;
355  }

358 IOReturn BasebandSPIDevice::exitLowPower( void )
359 {
360  SPI_TRACE( DEBUG_INFO );
361 
362  // always clear the flags
363  fLowPowerFlags &= ~ kFlagSystemInLowPower;
364  fCommandGate->commandWakeup( &fLowPowerFlags );
365 
366  switch ( fState )
367  {
368  case kStateLowPower:
369  {
370  IOReturn ret = fController->exitLowPower( this, fProtocol );
371  SPI_ASSERT_SUCCESS( ret );
372 
373  ret = fProtocol->exitLowPower();
374  SPI_ASSERT_SUCCESS( ret );
375 
376  fState = fStateBeforeLowPower;
377 
378  SPI_LOG( DEBUG_INFO, "Exited Low Power\n" );
379  }
380  return kIOReturnSuccess;
381 
382  default:
383  SPI_LOG( DEBUG_CRITICAL, kInvalidState, stateAsString( fState ) );
384 
385  return kIOReturnNotPermitted;
