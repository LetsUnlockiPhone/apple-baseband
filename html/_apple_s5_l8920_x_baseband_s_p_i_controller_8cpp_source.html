102 IOReturn AppleS5L8920XBasebandSPIController::loadConfiguration( void )
103 {
104  IOReturn ret;
105  ret = super::loadConfiguration();
106  if ( ret != kIOReturnSuccess )
107  {
108  return ret;
109  }
110 
111  _spiConReg = kS5L8920XSPCONClkSelNClk | kS5L8920XSPCONClkEnable | kS5L8920XSPCONMasterMode;
112 
113  if (fConfiguration.spiClockPolarity) _spiConReg |= kS5L8920XSPCONClkActiveLow;
114  if (fConfiguration.spiClockPhase) _spiConReg |= kS5L8920XSPCONClkMissingLast;
115  if (!fConfiguration.spiBitFirst) _spiConReg |= kS5L8920XSPCONBitOrderLSB;
116 
117  UInt32 sclkPeriod = fConfiguration.spiClockPeriod;
118  {
119  UInt32 period = getOverrideClockPeriod();
120  if ( period != 0 )
121  {
122  sclkPeriod = period;
123  SPI_LOG( DEBUG_CRITICAL, "Overriding SCLK to %uns\n", period );
124  }
125  }
126 
127  SPI_LOG( DEBUG_SPEW, "polarity %p, phase %p\n", fConfiguration.spiClockPolarity, fConfiguration.spiClockPhase );
128 
129  switch ( fConfiguration.spiWordSize )
130  {
131  case 8:
132  _spiWordSize = 0;
133  break;
134  case 16:
135  _spiWordSize = 1;
136  break;
137  case 32:
138  _spiWordSize = 2;
139  break;
140  default:
141  SPI_ASSERT( fConfiguration.spiWordSize, ==, 8 );
142  }
143 
144  _spiConReg |= _spiWordSize << kS5L8920XSPCONWordSizeShift;
145  _spiWordSize = 1 << _spiWordSize;
146 
147 #if SPI_WORD32
148  // Request in the unit of every one buffer. The documentation
149  // states one byte however it is wrong.
150  // TODO Is it worse to poll the last 4 buffers
151  _spiConReg |= kS5L8920XSPCONDMAReqOne;
152 #endif
153 
154  if ( !fConfiguration.spiBigEndian )
155  {
156  _spiConReg |= kS5L8920XSPCONDataSwap;
157  }
158 
159  // SPI Clock Value = Cell Clock / prescale
160  UInt64 tmp = (((UInt64)fNCLKFrequency * (UInt64)sclkPeriod) + 1000000000ULL - 1) / 1000000000ULL;
161 
162  _spiPreReg = tmp;
163 
164  SPI_LOG( DEBUG_CRITICAL, "NCLK Frequency %u, Prescaler %u\n", (unsigned)fNCLKFrequency, _spiPreReg );
165 
166  // SPI Inter-byte Delay = PCLK period * delay cycles
167  _spiIDDReg = (((UInt64)fPCLKFrequency * fConfiguration.spiWordDelay) + 1000000000ULL - 1) / 1000000000ULL;
168 
169  SPI_LOG( DEBUG_INFO, "SPCON: %p\tWord Size: %p\n", (void *)_spiConReg, (void *)_spiWordSize );
170  SPI_LOG( DEBUG_INFO, "SPPRE: %p\tSPIDD: %p\n", (void *)_spiPreReg, (void *)_spiIDDReg );
171 
172  return kIOReturnSuccess;
173 }

175 IOReturn AppleS5L8920XBasebandSPIController::configureHardware( UInt32 dmaFlags, IOByteCount size )
176 {
177  enableClock(true);
178  // Reset & Clear Interrupts
179  _spiClkConReg = kS5L8920XSPCLKCONEnable;
180 
181  // disable the clock
182  writeReg32(kS5L8920XSPCLKCON, kS5L8920XSPCLKCONFlushTx | kS5L8920XSPCLKCONFlushRx);
183  writeReg32(kS5L8920XSPPIN, 0);
184 
185  // Configure controller mode
186  writeReg32(kS5L8920XSPCNT, 0);
187  writeReg32(kS5L8920XSPTDCNT, 0);
188 
189  writeReg32(kS5L8920XSPPRE, _spiPreReg);
190  writeReg32(kS5L8920XSPIDD, _spiIDDReg);
191  writeReg32(kS5L8920XSPIRTO, 0xffffffff);
192  writeReg32(kS5L8920XSPIHANGD, 0);
193  writeReg32(kS5L8920XSPSTA, kS5L8920XSPSTAIntAll);
194 
195  size = size >> (_spiWordSize >> 1);
196  writeReg32( kS5L8920XSPCNT, size );
197 
198  UInt32 spiConReg = _spiConReg | kS5L8920XSPCONSPIModeDMA;
199  if ( dmaFlags & kTxDMA )
200  {
201  writeReg32( kS5L8920XSPTDCNT, size );
202  spiConReg |= kS5L8920XSPCONFullDuplexMode;
203  }
204  else
205  {
206  spiConReg |= kS5L8920XSPCONRxOnlyMode;
207  }
208 
209  // set the SPI controller register
210  writeReg32(kS5L8920XSPCON, spiConReg);
211 
212  return kIOReturnSuccess;
213 }
